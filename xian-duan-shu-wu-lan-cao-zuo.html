模板题

定义

线段树是基于分治思想的二叉树，用来维护区间信息（区间和、区间最值等），可以在 lognlognlogn 的时间内执行区间修改和区间查询

线段树中每个叶子节点存储元素本身，非叶子节点存储区间内元素的统计值

数据结构

节点数组tree[]tree[]tree[]，包括三个域，l,r,suml,r,suml,r,sum
至少开2N2N2N的数组。由于不一定是满二叉树，所以开4N4N4N

步骤

递归建树

父节点编号ppp 左子节点编号2p2p2p 右子节点编号2p+12p+12p+1
如果搜到叶子就返回，不是就从中间裂开各自往左右子树创建
离开时更新统计值

void create(int p,int l,int r){
	tree[p]={l,r,a[l]};//初始化
	if(l==r) return; //如果是叶子结点就返回 
	int mid=(l+r)/2,la=2*p,ra=2*p+1;//劈开 
	create(la,l,mid);//建左子树 
	create(ra,mid+1,r);//建右子树 sum;
	tree[p].sum=tree[la].sum+tree[ra].sum;//更新 
}

点更新

从根节点进入，递归找到叶子节点[x,x][x,x][x,x]，把该节点加上kkk，然后从上往下更新

void update(int p,int x,int v){ 
//先更新叶子 
	if(tree[p].l==x && tree[p].r==x){
		tree[p].sum+=v;
		return;
	}
	int mid=(tree[p].l+tree[p].r)/2,la=2*p,ra=2*p+1;//劈开
	//往下更新 
	if(x<=mid) update(la,x,v);
	if(x>mid) update(ra,x,v);
	tree[p].sum=tree[la].sum+tree[ra].sum;
}

区间查询

  • 若查询区间[x,y][x,y][x,y]完全覆盖该节点空间就返回sumsumsum
  • 若左子节点与[x,y][x,y][x,y]有重叠，则递归访问左子树
  • 若右子节点与[x,y][x,y][x,y]有重叠，则递归访问右子树
int find(int p,int l,int r){
	if(l<=tree[p].l && tree[p].r<=r) return tree[p].sum; //完全覆盖就返回查询值
	int mid=(tree[p].l+tree[p].r)/2,la=2*p,ra=2*p+1;//劈开 
	int sum=0;
	if(x<=mid) sum+=find(la,l,r);//访问左子树 
	if(y>mid) sum+=find(ra,l,r);//访问右子树 
	return sum; 
}

#include<iostream>
using namespace std;
const int N=5e5+10;
struct node{
	int l,r,sum;
}tree[4*N];
int a[N];
int n,m,op;
int x,y,k;
void create(int p,int l,int r){
	tree[p]={l,r,a[l]};//初始化
	if(l==r) return; //如果是叶子结点就返回 
	int mid=(l+r)/2,la=2*p,ra=2*p+1;//劈开 
	create(la,l,mid);//建左子树 
	create(ra,mid+1,r);//建右子树 sum;
	tree[p].sum=tree[la].sum+tree[ra].sum;//更新 
}
void update(int p,int x,int v){ 
//先更新叶子 
	if(tree[p].l==x && tree[p].r==x){
		tree[p].sum+=v;
		return;
	}
	int mid=(tree[p].l+tree[p].r)/2,la=2*p,ra=2*p+1;//劈开
	//往下更新 
	if(x<=mid) update(la,x,v);
	if(x>mid) update(ra,x,v);
	tree[p].sum=tree[la].sum+tree[ra].sum;
}
int find(int p,int l,int r){
	if(l<=tree[p].l && tree[p].r<=r) return tree[p].sum; //完全覆盖就返回查询值
	int mid=(tree[p].l+tree[p].r)/2,la=2*p,ra=2*p+1;//劈开 
	int sum=0;
	if(x<=mid) sum+=find(la,l,r);//访问左子树 
	if(y>mid) sum+=find(ra,l,r);//访问右子树 
	return sum; 
}
int main(){
	int n,m;
	cin>>n>>m;
	for(int i=1;i<=n;i++){
		cin>>a[i];
	}
	create(1,1,n);
	for(int i=1;i<=m;i++){
		cin>>op;
		if(op==1){
			cin>>x>>k;
			update(1,x,k);
		}else{
			cin>>x>>y;
			cout<<find(1,x,y)<<endl;
		}
	}
	return 0;
}
